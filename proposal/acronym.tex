% acronym command
\newcommand{\acronym}[2]{
  \DTLnewrow{acrdb}%
  \DTLnewdbentry{acrdb}{acronym}{#1}%
  \DTLnewdbentry{acrdb}{description}{#2}%
}

\newenvironment{acronyms}{%
  \DTLifdbexists{acrdb}{\DTLcleardb{acrdb}}{\DTLnewdb{acrdb}}%
}{%
  \DTLsort{acronym}{acrdb}%
  \DTLforeach*{acrdb}{\theAcr=acronym,\theDesc=description}
  {%
    \theAcr\dotfill\theDesc\\
  }%
}
